### Tables
```sql
CREATE TABLE IF NOT EXISTS `users` (
  `id` int(11) unsigned NOT NULL auto_increment,
  `username` varchar(50) NOT NULL,
  `hash` BINARY(32) NOT NULL,
  `isActive` BOOLEAN,
  PRIMARY KEY(`id`)
);

CREATE TABLE IF NOT EXISTS `koPos` (
  `id` int(11) unsigned NOT NULL auto_increment,
  `pos` nvarchar(400) NOT NULL,
  PRIMARY KEY(`id`)
);
```
